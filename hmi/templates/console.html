No page reloads occur; commands are sent asynchronously and responses are appended to the console output div in real time. ======================================================================== --> <script> /* -------------------------------------------------------------------------- DOM Element References -------------------------------------------------------------------------- Cache references to the console output div and the command input field so we don't have to query the DOM on every interaction. -------------------------------------------------------------------------- */ const output = document.getElementById('console-output'); const input = document.getElementById('cmd-input'); /* -------------------------------------------------------------------------- appendOutput(text) -------------------------------------------------------------------------- Appends a line of text to the console output area and automatically scrolls to the bottom so the most recent output is always visible. Uses textContent (not innerHTML) to safely display raw ATG responses without risk of HTML injection. -------------------------------------------------------------------------- */ function appendOutput(text) { output.textContent += '\n' + text; output.scrollTop = output.scrollHeight; } /* -------------------------------------------------------------------------- sendCmd() — Async Command Submission -------------------------------------------------------------------------- Reads the current value of the command input field, clears it, echoes the command to the console output (prefixed with ">"), and sends it to the server via an asynchronous POST request to /api/command. Request format: POST /api/command Content-Type: application/json Body: { "command": "<TLS command string>" } Expected response format (JSON): Success: { "response": "<raw ATG output text>" } Error: { "error": "<error description>" } If the fetch() call itself fails (e.g., network error, server down), the catch block displays a CONNECTION ERROR message in the console. Flow: 1. Read and trim the input value; abort if empty. 2. Clear the input field so the operator can type the next command. 3. Echo the command to the console output for reference. 4. POST the command to /api/command as JSON. 5. Parse the JSON response. 6. If data.error exists, display "ERROR: ..." in the console. 7. Otherwise, display the raw response text from data.response. 8. On network/fetch failure, display "CONNECTION ERROR: ..." instead. -------------------------------------------------------------------------- */ async function sendCmd() { const cmd = input.value.trim(); if (!cmd) return; input.value = ''; appendOutput('\n> ' + cmd); try { const resp = await fetch('/api/command', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({command: cmd}) }); const data = await resp.json(); if (data.error) { appendOutput('ERROR: ' + data.error); } else { appendOutput(data.response); } } catch (e) { appendOutput('CONNECTION ERROR: ' + e.message); } } /* -------------------------------------------------------------------------- quickCmd(cmd) — Quick-Access Button Handler -------------------------------------------------------------------------- Called by each quick-access button's onclick handler. Sets the input field's value to the pre-defined command string and immediately calls sendCmd() to execute it. This provides a one-click shortcut for common TLS commands without the operator needing to type anything. -------------------------------------------------------------------------- */ function quickCmd(cmd) { input.value = cmd; sendCmd(); } </script> {% endblock %}
